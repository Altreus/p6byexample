# Scalar ($)
my $name = 'Bob';
my Int $age = 12; # You can specify a type for the variable container

# Positional (@)
my @animals = 'Cow', 'Pig', 'Horse', 'Duck';

# Associative (%)
my %planet-mass = Earth => 1, Mars => 0.107, Jupiter => 318, Pluto => 0.002;

# Callable (&)
my &say-hello = sub { say "hello!" };
&say-hello();
